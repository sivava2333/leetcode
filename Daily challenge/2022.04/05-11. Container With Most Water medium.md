# 3. Code

```
class Solution {
    public int maxArea(int[] height) {
        int l = 0, r = height.length - 1;
        int ans = 0;
        while (l < r) {
            int border = Math.min(height[l], height[r]);
            ans = Math.max(ans,  border * (r - l));
            if (border == height[l]) {
                l++;
            } else {
                r--;
            }
        }
        
        return ans;
    }
}
```
