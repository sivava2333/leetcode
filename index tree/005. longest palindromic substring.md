## Manacher算法

### 代码
```
class Solution {
    public String longestPalindrome(String s) {
        if (s == null || s.length() == 0) {
            return "";
        }
        
        char[] str = manacherString(s.toCharArray());
        int[] pArr = new int[str.length];
        int C = -1;
        int R = -1;
        int max = Integer.MIN_VALUE;
        int index = -1;
        for (int i = 0; i < str.length; i++) {
            pArr[i] = i < R ? Math.min(pArr[2 * C - i], R - i) : 1;
            while (i + pArr[i] < str.length && i - pArr[i] > -1) {
                if (str[i + pArr[i]] == str[i - pArr[i]]) {
                    pArr[i]++;
                } else {
                    break;
                }
            }
            
            if (i + pArr[i] > R) {
                R = i + pArr[i];
                C = i;
            }
            
            if (pArr[i] > max) {
                index = i;
                max = pArr[i];
            }
        }
        
        int r = (max - 1) / 2;
        return (max & 1) == 0 ? s.substring(index - r, index + r + 1) : s.substring(index - r, index + r);
        
    }
    
    private char[] manacherString(char[] str) {
        char[] manacher = new char[str.length * 2 + 1];
        
        int index = 0;
        for (int i = 0; i < manacher.length; i++) {
            manacher[i] = (i & 1) == 0 ? '#' : str[index++];
        }
        
        return manacher;
    }
}
```
